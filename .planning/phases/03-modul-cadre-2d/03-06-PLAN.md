```tsx
import { useState, useEffect, useCallback } from 'react'
import { useLang } from '@/hooks/useLang'
import { useFrameSolver } from '@/hooks/useFrameSolver'
import { FrameToolbar } from '@/components/frame/FrameToolbar'
import { FrameCanvas } from '@/components/frame/FrameCanvas'
import { FrameResultTabs, exportFramePdf } from '@/components/frame/FrameResultTabs'
import { FrameDiagrams } from '@/components/frame/FrameDiagrams'
import { encodeFrameHash, decodeFrameHash } from '@/lib/frameHash'
import type { FrameInput } from '@/types/api'
import type { EditorMode } from '@/components/frame/FrameToolbar'
import type { DiagramTab } from '@/components/frame/FrameDiagrams'

const STORAGE_KEY = 'structcalc-frame-last'

const DEFAULT_INPUT: FrameInput = {
  nodes: [],
  bars: [],
  node_loads: [],
  bar_loads: [],
}

export default function FramePage() {
  const { t } = useLang()
  const { solve, result, loading, error, reset } = useFrameSolver()
  const [mode, setMode] = useState<EditorMode>('add_node')
  const [activeTab, setActiveTab] = useState<DiagramTab>('M')
  const [copySuccess, setCopySuccess] = useState(false)

  // State: priority → URL hash → localStorage → default (per D-19, D-20)
  const [input, setInput] = useState<FrameInput>(() => {
    const hashPayload = window.location.hash.replace(/^#/, '')
    const fromHash = decodeFrameHash(hashPayload)
    if (fromHash) return fromHash
    try {
      const stored = localStorage.getItem(STORAGE_KEY)
      if (stored) return JSON.parse(stored) as FrameInput
    } catch { /* ignore */ }
    return DEFAULT_INPUT
  })

  // On mount: auto-solve if hash is present (per D-19)
  useEffect(() => {
    const hashPayload = window.location.hash.replace(/^#/, '')
    const fromHash = decodeFrameHash(hashPayload)
    if (fromHash) {
      solve(fromHash)
    }
  }, []) // eslint-disable-line react-hooks/exhaustive-deps

  const handleSolve = useCallback(async () => {
    // Persist to localStorage (per D-20)
    localStorage.setItem(STORAGE_KEY, JSON.stringify(input))
    // Update URL hash (per D-19)
    window.location.hash = encodeFrameHash(input)
    await solve(input)
  }, [input, solve])

  const handleCopyLink = useCallback(() => {
    // Copy URL to clipboard (per D-21)
    navigator.clipboard.writeText(window.location.href).catch(() => {
      const el = document.createElement('input')
      el.value = window.location.href
      document.body.appendChild(el)
      el.select()
      document.execCommand('copy')
      document.body.removeChild(el)
    })
    setCopySuccess(true)
    setTimeout(() => setCopySuccess(false), 2000)
  }, [])

  const handleExportPdf = useCallback(() => {
    exportFramePdf(t('frame.page.title')).catch(console.error)
  }, [t])

  const canSolve = input.nodes.length >= 2 && input.bars.length >= 1

  void reset // available for future clear button

  return (
    <div className="min-h-screen pt-16">
      <div className="max-w-[1200px] mx-auto px-6 py-8">
        <div className="mb-2">
          <h1 className="text-3xl font-bold text-[var(--brand-text)]">
            {t('frame.page.title')}
          </h1>
        </div>
        <p className="text-[var(--brand-muted)] mb-8">{t('frame.page.subtitle')}</p>

        {/* Error banner */}
        {error && (
          <div
            role="alert"
            className="mb-6 p-4 rounded-lg bg-red-50 dark:bg-red-900/20 border border-red-200 dark:border-red-800 text-red-700 dark:text-red-300"
          >
            {error}
          </div>
        )}

        {/* Loading indicator */}
        {loading && (
          <div
            aria-busy="true"
            className="mb-6 h-2 rounded bg-[var(--brand-border)] animate-pulse"
          />
        )}

        {/* Toolbar (top, full width) */}
        <FrameToolbar
          mode={mode}
          onModeChange={setMode}
          onSolve={handleSolve}
          loading={loading}
          canSolve={canSolve}
        />

        {/* Canvas — FrameSidePanel is rendered internally by FrameCanvas */}
        <FrameCanvas
          mode={mode}
          onModeChange={setMode}
          nodes={input.nodes}
          bars={input.bars}
          nodeLoads={input.node_loads}
          barLoads={input.bar_loads}
          onNodesChange={(nodes) => setInput(prev => ({ ...prev, nodes }))}
          onBarsChange={(bars) => setInput(prev => ({ ...prev, bars }))}
          onNodeLoadsChange={(node_loads) => setInput(prev => ({ ...prev, node_loads }))}
          onBarLoadsChange={(bar_loads) => setInput(prev => ({ ...prev, bar_loads }))}
        />

        {/* Results section — only visible after successful solve */}
        {result && (
          <div id="frame-results-capture" className="mt-6">
            <FrameResultTabs
              activeTab={activeTab}
              onTabChange={setActiveTab}
              onExportPdf={handleExportPdf}
              onCopyLink={handleCopyLink}
              copySuccess={copySuccess}
            />
            <FrameDiagrams
              nodes={input.nodes}
              bars={input.bars}
              result={result}
              activeTab={activeTab}
            />
          </div>
        )}
      </div>
    </div>
  )
}
```
