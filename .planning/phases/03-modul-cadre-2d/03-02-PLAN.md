```python
from pydantic import BaseModel, field_validator, model_validator
from typing import List, Dict, Literal


# ── Input models ────────────────────────────────────────────────────────────

class FrameNode(BaseModel):
    id: str
    x: float
    y: float
    constraint: Literal['free', 'pin', 'roller', 'fixed'] = 'free'


class FrameBar(BaseModel):
    id: str
    node_i: str   # FrameNode.id
    node_j: str   # FrameNode.id
    EI: float = 21000.0    # kN·m²
    EA: float = 2100000.0  # kN

    @field_validator('EI', 'EA')
    @classmethod
    def must_be_positive(cls, v: float) -> float:
        if v <= 0:
            raise ValueError('EI and EA must be positive')
        return v


class NodeLoad(BaseModel):
    node_id: str
    Fx: float = 0.0   # kN, positive = rightward
    Fy: float = 0.0   # kN, positive = upward
    Mz: float = 0.0   # kN·m, positive = CCW


class BarLoad(BaseModel):
    bar_id: str
    q: float = 0.0         # kN/m, positive = downward (global Y-)
    q_start: float = 0.0   # fraction 0..1 of bar length
    q_end: float = 1.0     # fraction 0..1 of bar length


class FrameInput(BaseModel):
    nodes: List[FrameNode]
    bars: List[FrameBar]
    node_loads: List[NodeLoad] = []
    bar_loads: List[BarLoad] = []

    @model_validator(mode='after')
    def validate_structure(self) -> 'FrameInput':
        if len(self.nodes) < 2:
            raise ValueError('Un cadru necesita cel putin 2 noduri')
        if len(self.bars) < 1:
            raise ValueError('Un cadru necesita cel putin o bara')
        # At least one support must exist
        supports = [n for n in self.nodes if n.constraint != 'free']
        if not supports:
            raise ValueError('Cadrul nu are reazeme — adauga cel putin un pin, roller sau incastrare')
        node_ids = {n.id for n in self.nodes}
        for bar in self.bars:
            if bar.node_i not in node_ids:
                raise ValueError(f'Bara {bar.id}: node_i "{bar.node_i}" nu exista')
            if bar.node_j not in node_ids:
                raise ValueError(f'Bara {bar.id}: node_j "{bar.node_j}" nu exista')
        return self


# ── Result models ───────────────────────────────────────────────────────────

class FrameBarDiagram(BaseModel):
    bar_id: str
    M: List[float]   # bending moment at n equally-spaced points along bar
    V: List[float]   # shear force
    N: List[float]   # axial force


class FrameNodeResult(BaseModel):
    node_id: str
    ux: float        # horizontal displacement (m)
    uy: float        # vertical displacement (m)
    phi_z: float     # rotation (rad)


class FrameResult(BaseModel):
    bar_diagrams: List[FrameBarDiagram]
    node_results: List[FrameNodeResult]
    reactions: Dict[str, float]   # key: "node_{id}_Fx|Fy|Mz", value: kN or kN·m
    max_M: float
    max_V: float
    max_N: float
```

```python
import numpy as np
from anastruct import SystemElements
from app.schemas.frame import (
    FrameInput, FrameResult, FrameBarDiagram, FrameNodeResult
)
from typing import Dict, List


def solve_frame(data: FrameInput) -> FrameResult:
    """
    Rezolva un cadru 2D folosind FEM (anastruct SystemElements).
    Suporta bare inclinate, noduri cu 3 DOF (ux, uy, phi_z).
    Mapping UUID node IDs → anastruct consecutive integer IDs (1-indexed).
    """
    # Build anastruct node ID map: data.nodes[i].id → i+1
    node_id_map: Dict[str, int] = {}
    for i, node in enumerate(data.nodes):
        node_id_map[node.id] = i + 1

    # Build anastruct element ID map: data.bars[i].id → i+1
    bar_id_map: Dict[str, int] = {}
    for i, bar in enumerate(data.bars):
        bar_id_map[bar.id] = i + 1

    # Index nodes by id for quick lookup
    node_by_id = {n.id: n for n in data.nodes}

    ss = SystemElements()

    # Add elements (must be called before adding supports/loads)
    for bar in data.bars:
        n_i = node_by_id[bar.node_i]
        n_j = node_by_id[bar.node_j]
        ss.add_element(
            location=[[n_i.x, n_i.y], [n_j.x, n_j.y]],
            EI=bar.EI,
            EA=bar.EA,
        )

    # Add supports
    for node in data.nodes:
        if node.constraint == 'free':
            continue
        anastruct_nid = node_id_map[node.id]
        if node.constraint == 'pin':
            ss.add_support_hinged(node_id=anastruct_nid)
        elif node.constraint == 'roller':
            ss.add_support_roll(node_id=anastruct_nid, direction='x')
        elif node.constraint == 'fixed':
            ss.add_support_fixed(node_id=anastruct_nid)

    # Add node loads (concentrated forces and moments)
    for nl in data.node_loads:
        nid = node_id_map[nl.node_id]
        if abs(nl.Fx) > 1e-9 or abs(nl.Fy) > 1e-9:
            ss.point_load(node_id=nid, Fx=nl.Fx, Fy=nl.Fy)
        if abs(nl.Mz) > 1e-9:
            ss.moment_load(node_id=nid, Tz=nl.Mz)

    # Add bar distributed loads
    for bl in data.bar_loads:
        if abs(bl.q) < 1e-9:
            continue
        el_id = bar_id_map[bl.bar_id]
        # anastruct: negative q = downward (in global Y)
        ss.q_load(element_id=el_id, q=-bl.q, direction='y')

    # Solve
    ss.solve()

    # Extract per-bar diagrams
    bar_diagrams: List[FrameBarDiagram] = []
    for bar in data.bars:
        el_id = bar_id_map[bar.id]
        el = ss.element_map[el_id]
        npts = len(el.bending_moment)
        bar_diagrams.append(FrameBarDiagram(
            bar_id=bar.id,
            M=[float(v) for v in el.bending_moment],
            V=[float(v) for v in el.shear_force],
            N=list(np.linspace(float(el.N_1), float(el.N_2), npts)),
        ))

    # Extract nodal displacements
    node_results: List[FrameNodeResult] = []
    for node in data.nodes:
        nid = node_id_map[node.id]
        r = ss.get_node_results_system(node_id=nid)
        node_results.append(FrameNodeResult(
            node_id=node.id,
            ux=float(r['ux']),
            uy=float(r['uy']),
            phi_z=float(r.get('phi_z', r.get('Tz', 0.0))),
        ))

    # Extract reactions for support nodes
    reactions: Dict[str, float] = {}
    for node in data.nodes:
        if node.constraint == 'free':
            continue
        nid = node_id_map[node.id]
        r = ss.get_node_results_system(node_id=nid)
        reactions[f'node_{node.id}_Fx'] = float(r['Fx'])
        reactions[f'node_{node.id}_Fy'] = float(r['Fy'])
        reactions[f'node_{node.id}_Mz'] = float(r.get('Tz', 0.0))

    # Compute max values
    all_M = [v for bd in bar_diagrams for v in bd.M]
    all_V = [v for bd in bar_diagrams for v in bd.V]
    all_N = [v for bd in bar_diagrams for v in bd.N]

    return FrameResult(
        bar_diagrams=bar_diagrams,
        node_results=node_results,
        reactions=reactions,
        max_M=float(max(abs(v) for v in all_M)) if all_M else 0.0,
        max_V=float(max(abs(v) for v in all_V)) if all_V else 0.0,
        max_N=float(max(abs(v) for v in all_N)) if all_N else 0.0,
    )
```

import pytest
from app.services.frame_solver import solve_frame
from app.schemas.frame import FrameInput, FrameNode, FrameBar, NodeLoad, BarLoad

# Portal frame fixture: 2 columns (h=3m) + 1 beam (L=4m)
# n1(0,0,pin) → n2(0,3,free) → n3(4,3,free) ← n4(4,0,roller)
# Load: Fx=10kN at n2

@pytest.fixture
def portal_input():
    return FrameInput(
        nodes=[
            FrameNode(id='n1', x=0.0, y=0.0, constraint='pin'),
            FrameNode(id='n2', x=0.0, y=3.0, constraint='free'),
            FrameNode(id='n3', x=4.0, y=3.0, constraint='free'),
            FrameNode(id='n4', x=4.0, y=0.0, constraint='roller'),
        ],
        bars=[
            FrameBar(id='b1', node_i='n1', node_j='n2', EI=21000.0, EA=2100000.0),
            FrameBar(id='b2', node_i='n2', node_j='n3', EI=21000.0, EA=2100000.0),
            FrameBar(id='b3', node_i='n4', node_j='n3', EI=21000.0, EA=2100000.0),
        ],
        node_loads=[NodeLoad(node_id='n2', Fx=10.0, Fy=0.0, Mz=0.0)],
        bar_loads=[],
    )


def test_portal_reactions_horizontal_equilibrium(portal_input):
    result = solve_frame(portal_input)
    fx_sum = sum(v for k, v in result.reactions.items() if k.endswith('_Fx'))
    # Sum of horizontal reactions must balance Fx=10 kN
    assert abs(abs(fx_sum) - 10.0) < 0.5, f"Fx sum = {fx_sum}, expected ~10.0"


def test_portal_vertical_reactions_zero_net(portal_input):
    result = solve_frame(portal_input)
    fy_sum = sum(v for k, v in result.reactions.items() if k.endswith('_Fy'))
    # No vertical load → net Fy reactions must sum to ~0
    assert abs(fy_sum) < 0.5, f"Fy sum = {fy_sum}, expected ~0"


def test_portal_max_M_positive(portal_input):
    result = solve_frame(portal_input)
    assert result.max_M > 0, f"max_M should be positive, got {result.max_M}"


def test_portal_bar_diagrams_count(portal_input):
    result = solve_frame(portal_input)
    assert len(result.bar_diagrams) == 3
    assert all(bd.bar_id in {'b1', 'b2', 'b3'} for bd in result.bar_diagrams)


def test_portal_node_results_count(portal_input):
    result = solve_frame(portal_input)
    assert len(result.node_results) == 4
    assert all(nr.node_id in {'n1', 'n2', 'n3', 'n4'} for nr in result.node_results)


def test_portal_diagram_lists_nonempty(portal_input):
    result = solve_frame(portal_input)
    for bd in result.bar_diagrams:
        assert len(bd.M) > 0
        assert len(bd.V) > 0
        assert len(bd.N) > 0
        assert len(bd.M) == len(bd.V) == len(bd.N)
