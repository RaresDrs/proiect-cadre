```typescript
  )

describe('BeamInputForm — REQ-02-01', () => {
```

**Step 2 — Create BeamInputForm.tsx (per D-04, D-05, D-06, D-08):**

Props interface:
```typescript
interface BeamInputFormProps {
  initialInput: BeamInput
  onSolve: (input: BeamInput) => void
  loading?: boolean
}
```

Structure:
- Controlled component: local `useState<BeamInput>` initialized from `initialInput`
- Top section: `length` (number input, >0, meters), `angle_deg` (number input, default 0), `EI` (number, default 21000), `EA` (number, default 2100000)
- Supports section: dynamic list — each row has `x` (number input) + `type` (Select: 1=Articulație/Pin, 2=Reazem mobil/Roller, 3=Încastrare/Fixed) + remove button. "Adaugă reazem / Add support" button appends `{ x: 0, type: 1 }`.
- Point loads section: dynamic list — each row has `x`, `fx`, `fy` (all number inputs) + remove button. "Adaugă forță / Add load" button appends `{ x: 0, fx: 0, fy: 0 }`.
- Distributed load section: `distributed_load` (kN/m), `q_start`, `q_end` number inputs.
- Stability check function (inline):
  ```typescript
  function isStable(supports: Support[]): boolean {
    if (supports.length < 2) return false
    // At minimum need 1 pin or fixed to resist horizontal, and total DOF constrained
    const hasHorizontalRestraint = supports.some(s => s.type === 1 || s.type === 3)
    const totalRestraints = supports.reduce((acc, s) => {
      if (s.type === 1) return acc + 2  // pin: Fx + Fy
      if (s.type === 2) return acc + 1  // roller: Fy only
      if (s.type === 3) return acc + 3  // fixed: Fx + Fy + M
      return acc
    }, 0)
    return hasHorizontalRestraint && totalRestraints >= 3
  }
  ```
- Submit: `<button>` disabled when `!isStable(input.supports) || loading`. Shows `t('beam.form.unstable')` warning text when not stable and supports.length > 0.
- All labels use `t('beam.form.*')` keys. Never hardcode strings.

Prop `onSolve` is called on button click with the current input state. This component does NOT call the API — that happens in BeamPage via `useBeamSolver` (02-02).

**Step 3 — Create BeamPreview.tsx (per D-07 — SVG live preview, debounced 150ms):**

Props:
```typescript
interface BeamPreviewProps {
  input: BeamInput
}
```

Implementation pattern — pure React SVG (NOT D3, D3 is only for M/T/N diagrams in 02-03):
- `useEffect` with a 150ms `setTimeout` debounce that re-renders the SVG whenever `input` changes
- SVG viewBox="0 0 600 200", responsive with `width="100%"`
- Draws:
  - Horizontal beam line from x=40 to x=560 (mapped from 0 to input.length)
  - For each support: small triangle (pin/roller) or hatched box (fixed) at the correct x position. Use simple SVG `<polygon>` for pin (triangle pointing up, vertex at beam), `<polygon>` for roller (triangle + circle underneath), `<rect>` with hatch pattern for fixed.
  - For each point load: arrow `<line>` with arrowhead `<polygon>`. Fy: vertical arrow (up if positive, down if negative). Fx: horizontal arrow.
  - For distributed load: multiple short vertical arrows between q_start and q_end, with a horizontal top line.
- Colors: `var(--brand-accent)` for beam, `var(--brand-primary)` for supports, `var(--color-success, #22c55e)` for loads — uses CSS variables so dark mode works automatically.

```typescript
// Debounce pattern:
const [debouncedInput, setDebouncedInput] = useState(input)
useEffect(() => {
  const t = setTimeout(() => setDebouncedInput(input), 150)
  return () => clearTimeout(t)
}, [input])
```
